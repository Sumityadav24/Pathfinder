<script>
const searchInput = document.getElementById('searchInput');
const generateBtn = document.getElementById('generateBtn');
const topicButtons = document.querySelectorAll('.topic-btn');
const roadmapContainer = document.getElementById('roadmapContainer');
const welcomeMessage = document.getElementById('welcomeMessage');

generateBtn.addEventListener('click', () => {
const query = searchInput.value.trim();
if (query) {
generateRoadmap(query);
} else {
displayError("Please enter a topic to generate a roadmap.");
}
});

searchInput.addEventListener('keypress', (event) => {
if (event.key === 'Enter') {
generateBtn.click();
}
});

topicButtons.forEach(button => {
button.addEventListener('click', () => {

const topic = button.dataset.topic;
searchInput.value = topic;
generateRoadmap(topic);
});
});

function toggleAccordion(header) {
const content = header.nextElementSibling;
const icon = header.querySelector('.icon');

content.classList.toggle('max-h-0');
content.classList.toggle('max-h-screen');
content.classList.toggle('pt-4');
content.classList.toggle('pb-4');
icon.classList.toggle('rotate-180');
}

function displayLoading() {
if (welcomeMessage) welcomeMessage.style.display = 'none';
roadmapContainer.innerHTML = `
<div class="flex flex-col items-center justify-center bg-white p-10 rounded-lg
shadow-md">
<div class="loader"></div>
<p class="mt-4 text-stone-600 font-medium">Crafting your learning path...</p>
<p class="mt-2 text-sm text-stone-500">This might take a moment.</p>
</div>
`;
}

function displayError(message) {
if (welcomeMessage) welcomeMessage.style.display = 'none';
roadmapContainer.innerHTML = `
<div class="text-center py-10 px-6 bg-red-50 border border-red-200 rounded-lg
shadow-md">
<div class="text-5xl mb-4">⚠️</div>
<h2 class="text-2xl font-semibold text-red-700">Oops! Something went
wrong.</h2>
<p class="text-red-600 mt-2">${message}</p>
</div>
`;
}

async function generateRoadmap(query) {
displayLoading();

const prompt = `
Create a detailed, structured learning roadmap for the topic: "${query}".

The roadmap should be broken down into logical steps from beginner to
advanced.
For each step, provide:
1. A clear "title".
2. A "description" explaining the goal of the step.
3. A list of key "topics" to learn.
4. A list of "practical_projects" or exercises to apply the knowledge.
5. A list of "resources" with placeholder names or types (e.g., 'Official
Documentation', 'Recommended YouTube Channel', 'Popular Online Course').

Return the output as a JSON object with a single key "roadmap" which is an array
of these steps.
Example step format:
{
"title": "Step 1: The Basics",
"description": "Start with the fundamentals.",
"topics": ["Topic A", "Topic B"],
"practical_projects": ["Project A"],
"resources": ["Resource A"]
}
`;

let chatHistory = [{ role: "user", parts: [{ text: prompt }] }];
const payload = {
contents: chatHistory,
generationConfig: {
responseMimeType: "application/json",
responseSchema: {
type: "OBJECT",
properties: {
"roadmap": {
type: "ARRAY",
items: {
type: "OBJECT",
properties: {
"title": { "type": "STRING" },
"description": { "type": "STRING" },
"topics": { "type": "ARRAY", "items": { "type": "STRING" } },
"practical_projects": { "type": "ARRAY", "items": { "type": "STRING"
} },
"resources": { "type": "ARRAY", "items": { "type": "STRING" } },
},
required: ["title", "description", "topics", "practical_projects",
"resources"]
}
}
}
}

}
};

const apiKey = ""; // Keep this empty
const apiUrl =
`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:
generateContent?key=${apiKey}`;

try {
const response = await fetch(apiUrl, {
method: 'POST',
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify(payload)
});

if (!response.ok) {
throw new Error(`API error! status: ${response.status}`);
}

const result = await response.json();

if (result.candidates && result.candidates[0].content &&
result.candidates[0].content.parts[0]) {
const jsonText = result.candidates[0].content.parts[0].text;
const data = JSON.parse(jsonText);
renderRoadmap(data.roadmap, query);
} else {
console.error("Unexpected API response structure:", result);
displayError("Could not parse the roadmap from the AI's response. The
structure might have been unexpected.");
}
} catch (error) {
console.error("Error fetching roadmap:", error);
displayError("Failed to generate the roadmap. Please check your connection or try
a different topic.");
}
}

function renderRoadmap(steps, query) {
if (!steps || steps.length === 0) {
displayError(`No roadmap could be generated for "${query}". Please try a broader
topic.`);
return;
}

let html = `<h2 class="text-3xl font-bold text-center mb-8">Learning Path:
${query}</h2><div class="space-y-4">`;

steps.forEach((step, index) => {
html += `
<div class="bg-white rounded-lg shadow-md overflow-hidden border
border-stone-200">
<div class="accordion-header cursor-pointer p-5 flex justify-between
items-center bg-stone-50 hover:bg-stone-100" onclick="toggleAccordion(this)">
<h3 class="text-xl font-semibold text-teal-800">${step.title}</h3>
<div class="icon transition-transform duration-300">
<svg xmlns="http://www.w3.org/2000/svg" class="h-6 w-6 text-teal-600"
fill="none" viewBox="0 0 24 24" stroke="currentColor">
<path stroke-linecap="round" stroke-linejoin="round" stroke-width="2"
d="M19 9l-7 7-7-7" />
</svg>
</div>
</div>
<div class="accordion-content max-h-0 overflow-hidden transition-all
duration-500 ease-in-out px-5">
<p class="text-stone-600 mb-6">${step.description || ''}</p>

<div class="grid md:grid-cols-3 gap-6">
${generateSectionHtml('�� Key Topics', step.topics)}
${generateSectionHtml('��️ Practical Projects', step.practical_projects)}
${generateSectionHtml('�� Resources', step.resources)}
</div>
</div>
</div>
`;
});

html += `</div>`;
roadmapContainer.innerHTML = html;
}

function generateSectionHtml(title, items) {
if (!items || items.length === 0) {
return '';
}
let listItems = items.map(item => `<li class="flex items-start"><span class="mr-2 mt-1
text-teal-500 flex-shrink-0">&#10003;</span><span>${item}</span></li>`).join('');
return `
<div class="bg-stone-50/50 p-4 rounded-md">
<h4 class="font-bold text-stone-700 mb-2">${title}</h4>
<ul class="space-y-2 text-stone-600">${listItems}</ul>
</div>
`;
}
